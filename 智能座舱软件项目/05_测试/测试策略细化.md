```kotlin
class TestMessageBuilder {
    private var id: String = UUID.randomUUID().toString()
    private var priority: MessagePriority = MessagePriority.P2_NORMAL
    private var category: MessageCategory = MessageCategory.GENERAL
    private var title: String = "测试消息"
    private var content: String = "测试内容"
    private var timestamp: Long = System.currentTimeMillis()
    private var expiryTime: Long? = null
    private var sender: String = "system"
    
    fun withId(id: String) = apply { this.id = id }
    fun withPriority(priority: MessagePriority) = apply { this.priority = priority }
    fun withCategory(category: MessageCategory) = apply { this.category = category }
    fun withTitle(title: String) = apply { this.title = title }
    fun withContent(content: String) = apply { this.content = content }
    fun withExpiry(timeMs: Long) = apply { this.expiryTime = timeMs }
    fun withSender(sender: String) = apply { this.sender = sender }
    fun withTimestamp(timestamp: Long) = apply { this.timestamp = timestamp }
    
    fun asSafetyMessage() = apply {
        this.category = MessageCategory.SAFETY
        this.priority = MessagePriority.P0_CRITICAL
    }
    
    fun asExpired() = apply {
        this.expiryTime = System.currentTimeMillis() - 1000
    }
    
    fun build(): Message {
        return Message(
            id = id,
            priority = priority,
            category = category,
            title = title,
            content = content,
            timestamp = timestamp,
            expiryTime = expiryTime,
            sender = sender
        )
    }
}
```

```kotlin
@DisplayName("驾驶安全控制器测试 - ASIL A")
class DrivingSafetyControllerTest {
    
    @MockK
    private lateinit var vehicleStateMonitor: VehicleStateMonitor
    
    @MockK
    private lateinit var popupManager: MessagePopupManager
    
    @MockK
    private lateinit var messageQueue: MessageQueue
    
    @InjectMockKs
    private lateinit var controller: DrivingSafetyController
    
    @BeforeEach
    fun setUp() {
        MockKAnnotations.init(this)
    }
    
    @Test
    @DisplayName("应当阻塞非安全弹窗当行驶速度≥15km/h")
    fun `should block non-safety popup when driving at or above 15 km_h`() {
        // Arrange: 模拟行驶状态
        every { vehicleStateMonitor.getCurrentSpeed() } returns 15
        every { vehicleStateMonitor.getCurrentGear() } returns GearPosition.DRIVE
        
        val entertainmentMessage = TestMessageMother
            .createHighPriorityMessage()
            .copy(category = MessageCategory.ENTERTAINMENT)
        
        // Act
        val result = controller.shouldShowPopup(entertainmentMessage)
        
        // Assert
        assertFalse(result.isAllowed)
        assertEquals(BlockReason.DRIVING_RESTRICTION, result.blockReason)
        verify { messageQueue.enqueue(entertainmentMessage) }
        verify(exactly = 0) { popupManager.show(any()) }
    }
    
    @Test
    @DisplayName("应当允许安全弹窗即使正在行驶")
    fun `should allow safety popup even when driving`() {
        every { vehicleStateMonitor.getCurrentSpeed() } returns 80
        every { vehicleStateMonitor.getCurrentGear() } returns GearPosition.DRIVE
        
        val safetyMessage = TestMessageMother
            .createHighPriorityMessage()
            .copy(category = MessageCategory.SAFETY)
        
        val result = controller.shouldShowPopup(safetyMessage)
        
        assertTrue(result.isAllowed)
        verify { popupManager.show(safetyMessage) }
    }
    
    // ASIL A要求：故障注入测试
    @Test
    @DisplayName("应当在车辆状态获取失败时进入故障安全模式")
    fun `should enter fail-safe mode when vehicle state unavailable`() {
        every { vehicleStateMonitor.getCurrentSpeed() } throws 
            VehicleStateException("CAN bus error")
        
        val message = TestMessageMother.createHighPriorityMessage()
        
        val result = controller.shouldShowPopup(message)
        
        // 故障安全：禁止显示
        assertFalse(result.isAllowed)
        assertEquals(BlockReason.SAFETY_SYSTEM_ERROR, result.blockReason)
    }
}
```
